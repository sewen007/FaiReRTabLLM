def rank_applicants(applicants):
    ranked_applicants = []
    for applicant in applicants:
        student_id = applicant[0].split()[-1]
        gender = 'female' if 'female' in applicant[0] else'male'
        ugpa = float(applicant[0].split()[2])
        lsat = float(applicant[0].split()[-1].split('(')[1][:-1])
        total_score = ugpa + lsat
        ranked_applicants.append({"student_id": int(student_id), "gender": gender})
    ranked_applicants.sort(key=lambda x: (-x['total_score'], -x['ugpa'], -x['lsat']))
    return ranked_applicants
