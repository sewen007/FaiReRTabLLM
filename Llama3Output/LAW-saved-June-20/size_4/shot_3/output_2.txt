import json

# Define the weights for the UGPA and LSAT scores
ugpa_weight = 0.6
lsat_weight = 0.4

    {"student_id": 14853, "gender": "female", "ugpa": 3.8, "lsat": 42.5}
]

# Calculate the weighted average score for each applicant
ranked_applicants = []
for applicant in applicants:
    weighted_average = ugpa_weight * applicant["ugpa"] + lsat_weight * applicant["lsat"]
    ranked_applicants.append({"student_id": applicant["student_id"], "gender": applicant["gender"]})

# Sort the applicants by their weighted average score
