from typing import List
from operator import itemgetter

def rank_students(students: List[List[str]]) -> List[dict]:
    ranked_students = sorted(students, key=lambda x: (float(x[2].split()[1]), float(x[3].split()[1])), reverse=True)
    return [{"student_id": int(x[0].split()[2]), "gender": x[1]} for x in ranked_students]
