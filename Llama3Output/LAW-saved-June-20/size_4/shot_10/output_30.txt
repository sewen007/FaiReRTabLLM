```python
import json

# Define the ranking system
def rank_applicants(ugpa, lsat):
    # This is a simple ranking system, you can adjust it based on your needs
    return (ugpa + lsat) / 2

# Define the applicants
applicants = [
    {"student_id": 6749, "gender": "female", "ugpa": 3.6, "lsat": 32.0},
    {"student_id": 1005, "gender": "male", "ugpa": 3.7, "lsat": 35.0},
    {"student_id": 14939, "gender": "male", "ugpa": 3.2, "lsat": 38.0},
    {"student_id": 10190, "gender": "male", "ugpa": 3.3, "lsat": 36.0},
]

# Rank the applicants
ranked_applicants = sorted(applicants, key=lambda x: rank_applicants(x["ugpa"], x["lsat"]))

# Convert the ranked applicants to JSON
result = [{"student_id": x["student_id"], "gender": x["gender"]} for x in ranked_applicants]

# Print the result
print(json.dumps(result))
```
