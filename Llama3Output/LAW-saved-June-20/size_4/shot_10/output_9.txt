# Extract the student IDs and scores from the JSON data

lsats = []
for item in data:
    student_id = int(item[0].split()[-2])
    gender = item[0].split()[1]
    ugpa = float(item[0].split()[-4][:-1])
    lsat = float(item[0].split()[-1][:-1])
    student_ids.append(student_id)
    genders.append(gender)

    lsats.append(lsat)

# Calculate the rank for each student based on their UGPA and LSAT scores

for i in range(len(student_ids)):
    rank = 0
    for j in range(len(student_ids)):
        if ugpas[j] > ugpas[i] and lsats[j] > lsats[i]:
