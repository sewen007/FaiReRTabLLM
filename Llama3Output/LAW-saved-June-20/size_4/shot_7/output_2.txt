```Python
def rank_applicants(applicants):
    ranked_applicants = []
    for applicant in applicants:
        id = applicant[0]
        gender = applicant[1]
        ugpa = float(applicant[2])
        lsat = float(applicant[3])
        weighted_score = (0.6 * ugpa) + (0.4 * lsat)
        ranked_applicants.append({"student_id": id, "gender": gender})
    ranked_applicants.sort(key=lambda x: (-x['student_id'], -x['gender'], -x['weighted_score']))
    return ranked_applicants
```
