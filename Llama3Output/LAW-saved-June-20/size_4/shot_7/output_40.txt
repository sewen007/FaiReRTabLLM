from scipy.stats import pearsonr
import numpy as np

def rank_applicants(applicants):
    ugpas = [float(applicant[2]) for applicant in applicants]
    lsats = [float(applicant[3]) for applicant in applicants]
    ranked_applicants = []
    for i in range(len(applicants)):
        for j in range(i+1, len(applicants)):
            if (ugpas[i] > ugpas[j]) or (ugpas[i] == ugpas[j] and lsats[i] > lsats[j]):
                ugpas[i], ugpas[j] = ugpas[j], ugpas[i]
                lsats[i], lsats[j] = lsats[j], lsats[i]
    for i in range(len(applicants)):
        ranked_applicants.append({"student_id": int(applicants[i][0]), "gender": applicants[i][1]})
    return ranked_applicants
