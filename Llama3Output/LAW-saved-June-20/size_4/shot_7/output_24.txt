def rank_applicants(applicants):
    ranked_applicants = []
    for applicant in applicants:
        student_id = int(applicant[0].split()[-1])
        gender = applicant[0].split()[1]
        ugpa = float(applicant[2].split(':')[1])
        lsat = float(applicant[3].split(':')[1])
        combined_score = ugpa + lsat
        ranked_applicants.append({"student_id": student_id, "gender": gender})
    ranked_applicants.sort(key=lambda x: (-x['combined_score'], -x['ugpa'], -x['lsat']))
    return ranked_applicants

applicants = [
    ['The student ID is 13938. The gender is female. The UGPA score is 3.6. The LSAT score is 41.0.'],

print(rank_applicants(applicants))
