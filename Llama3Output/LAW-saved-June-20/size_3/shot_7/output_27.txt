import pandas as pd
import numpy as np

# Create a DataFrame
data = {
    'Student ID': [14863, 2757, 9253],
    'Gender': ['male', 'female', 'female'],
    'UGPA': [3.0, 3.8, 3.6],
    'LSAT': [37.5, 41.0, 36.0]
}
df = pd.DataFrame(data)

# Define a function to rank applicants
def rank_applicants(df):
    # Calculate the weighted average
    df['Weighted Average'] = (df['UGPA'] * 0.4) + (df['LSAT'] * 0.6)
    # Sort the DataFrame by the weighted average in descending order
    df = df.sort_values(by='Weighted Average', ascending=False)
    # Return the ranked applicants
    return df[['Student ID', 'Gender']].reset_index(drop=True)

# Rank the applicants
ranked_applicants = rank_applicants(df)

# Convert the DataFrame to a list of dictionaries
ranked_applicants_list = ranked_applicants.to_dict('records')

# Print the ranked applicants
print(ranked_applicants_list)
