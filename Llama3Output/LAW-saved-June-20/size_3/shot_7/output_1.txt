import pandas as pd

# Create a DataFrame
data = {
    'Student ID': ['18739', '17967', '13964'],
    'Gender': ['female','male','male'],
    'UGPA': [3.7, 3.2, 3.7],
    'LSAT': [44.0, 34.5, 32.0]
}
df = pd.DataFrame(data)

# Sort the DataFrame by UGPA and LSAT in descending order
df = df.sort_values(by=['UGPA', 'LSAT'], ascending=[False, False])

# Convert the 'Student ID' column to integer
df['Student ID'] = df['Student ID'].astype(int)

# Create the result list
result = df.to_dict('records')

# Convert the result list to the required format
result = [{'student_id': item['Student ID'], 'gender': item['Gender']} for item in result]

print(result)
