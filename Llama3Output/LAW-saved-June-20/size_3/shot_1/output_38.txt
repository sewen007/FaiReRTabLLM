def rank_applicants(applicants):
    ranked_applicants = []
    for applicant in applicants:
        student_id, gender, ugpa, lsat = applicant[0].split()[2], applicant[0].split()[1], float(applicant[0].split()[4]), float(applicant[0].split()[6])
        ranked_applicants.append({"student_id": int(student_id.split(":")[1]), "gender": gender})
    ranked_applicants.sort(key=lambda x: (x['ugpa'], x['lsat']), reverse=True)
    return ranked_applicants

applicants = []
for applicant in [[ 'The student ID is 17657. The gender is female. The UGPA score is 2.8. The LSAT score is 30.5.'],
                   ['The student ID is 21394. The gender is female. The UGPA score is 3.3. The LSAT score is 43.0.'],
                   ['The student ID is 360. The gender is female. The UGPA score is 2.7. The LSAT score is 27.0.']
                  ]:
    applicants.append(applicant[0].split('. ')[1:])

print(rank_applicants(applicants))
