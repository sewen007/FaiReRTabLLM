def rank_applicants(applicants):
    # Define the ranking criteria
    ranking_criteria = [
        ("LSAT", 1),
        ("UGPA", 0.5)
    ]

    # Define the ranking function
    def ranking_function(applicant):
        return sum(score * weight for score, weight in ranking_criteria)

    # Rank the applicants
    ranked_applicants = sorted(applicants, key=ranking_function, reverse=True)

    # Return the ranked results in the specified JSON format
    return [{"student_id": applicant[0], "gender": applicant[1]} for applicant in ranked_applicants]
