def rank_students(students):
    ranked_students = sorted(students, key=lambda x: (x[3], x[2]), reverse=True)
    result = []
    for student in ranked_students:
        result.append({"student_id": student[0], "gender": student[1]})
    return result

students = [
    [13938, "female", 3.6, 41.0],
    [8820, "female", 2.9, 41.0],
    [5876, "male", 3.0, 38.5]
]

print(rank_students(students))
