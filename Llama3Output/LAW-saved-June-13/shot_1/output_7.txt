def rank_students(students):
    ranked_students = []
    for student in students:
        student_id = student[0]
        gender = student[1]
        ugpa = float(student[2])
        lsat = float(student[3])
        weighted_score = 0.4 * ugpa + 0.6 * lsat
        ranked_students.append({"student_id": student_id, "gender": gender})
    ranked_students.sort(key=lambda x: (x["student_id"], -x["gender"], -x["student_id"]), reverse=True)
    return ranked_students
