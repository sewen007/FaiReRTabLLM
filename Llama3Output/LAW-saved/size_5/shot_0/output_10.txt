# Import necessary libraries
import json
from operator import itemgetter

# Define the applicants
applicants = [
    {'student_id': 9074, 'gender': 'female', 'UGPA': 3.0, 'LSAT': 41.0},
    {'student_id': 7704, 'gender': 'female', 'UGPA': 3.6, 'LSAT': 37.0},
    {'student_id': 10757, 'gender': 'female', 'UGPA': 3.6, 'LSAT': 33.0},
    {'student_id': 17408, 'gender': 'female', 'UGPA': 3.4, 'LSAT': 37.0},
    {'student_id': 13770, 'gender': 'female', 'UGPA': 2.9, 'LSAT': 39.0}
]

# Sort the applicants by UGPA and LSAT scores
applicants.sort(key=itemgetter('UGPA', 'LSAT'), reverse=True)

# Create a list to store the ranked applicants
ranked_applicants = []

# Iterate over the sorted applicants
for i, applicant in enumerate(applicants):
    ranked_applicants.append({'student_id': applicant['student_id'], 'gender': applicant['gender']})

# Convert the list to a JSON object
ranked_applicants_json = json.dumps(ranked_applicants)

# Print the ranked applicants in JSON format
print(ranked_applicants_json)
