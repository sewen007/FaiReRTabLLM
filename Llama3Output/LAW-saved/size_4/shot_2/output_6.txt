def rank_applicants(applicants):
    ranked_applicants = sorted(applicants, key=lambda x: (-x['UGPA'], -x['LSAT']))
    ranked_applicants_json = []
    for applicant in ranked_applicants:
        ranked_applicants_json.append({"student_id": applicant['Student ID'], "gender": applicant['Gender']})
    return ranked_applicants_json
