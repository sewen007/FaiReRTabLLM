```Python
def rank_applicants(applicants):
    ranked_applicants = sorted(applicants, key=lambda x: (-x['UGPA'], -x['LSAT']))
    return [{"student_id": applicant['Student ID'], "gender": applicant['Gender']} for applicant in ranked_applicants]

applicants = [
    {"Student ID": 19861, "Gender": "male", "UGPA": 3.0, "LSAT": 39.0},
    {"Student ID": 10950, "Gender": "female", "UGPA": 3.0, "LSAT": 40.0},

    {"Student ID": 16299, "Gender": "male", "UGPA": 3.6, "LSAT": 37.5}
]

print(rank_applicants(applicants))
```
