```python
import json
import operator

def rank_applicants(applicants):
    ranked_applicants = sorted(applicants.items(), key=operator.itemgetter(1, 0), reverse=True)
    ranked_applicants_json = [{"student_id": student[0], "gender": student[1][0]} for student in ranked_applicants]
    return json.dumps(ranked_applicants_json)

applicants = {
    6465: ("female", 2.7, 45.0),
    14842: ("female", 3.3, 39.0),
    3647: ("female", 2.8, 28.0),
    21192: ("female", 3.1, 42.0)
}

print(rank_applicants(applicants))
```
