import json

def rank_applicants(applicants):
    ranked_applicants = sorted(applicants, key=lambda x: (-x['LSAT'], -x['UGPA'], -x['Student ID']))
    return [{"student_id": x['Student ID'], "gender": x['Gender']} for x in ranked_applicants]

# test the function
applicants = [
    {"Student ID": 19283, "Gender": "male", "UGPA": 3.6, "LSAT": 38.0},
    {"Student ID": 6126, "Gender": "male", "UGPA": 3.6, "LSAT": 29.5},
    {"Student ID": 18494, "Gender": "female", "UGPA": 3.0, "LSAT": 30.0},
    {"Student ID": 10926, "Gender": "male", "UGPA": 3.2, "LSAT": 38.0}
]

print(json.dumps(rank_applicants(applicants)))
